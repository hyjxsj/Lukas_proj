$(function(){    

// 浏览记录
function createStorageForHistory () {

    // 存储storage
    function setStorage (n,t) {
        if (typeof t != 'string')  t = JSON.stringify(t)
        window.localStorage.setItem(n,t)
    }
    // 获取storage
    function getStorage (n) {
        return JSON.parse(window.localStorage.getItem(n))
    }

    // 首次调用localStorage时放入空数组
    var emptyStorage = []
    
    // 查找localStorage中是否含有浏览记录数组
    if (localStorage.hasOwnProperty('historyLog')) {
        var oldStorage = getStorage('historyLog')
        if (typeof markettitle == 'undefined' && typeof marketurl == 'undefined') {
            markettitle = 0
            marketurl = 0
        } else {
            var itemObj = {
                tit : markettitle,
                url : marketurl
            }
            oldStorage.unshift(itemObj)
        }
        // 判断浏览记录中是否有相同记录 有则删除
        for (var i = 1; i < oldStorage.length; i++) {
            if (oldStorage[0].tit == oldStorage[i].tit) {
                oldStorage.splice(i,1)
                setStorage('historyLog',oldStorage)
            }
        }
        // 限制浏览记录不超过5条
        if (oldStorage.length > 5) {
            oldStorage.pop()
            setStorage('historyLog',oldStorage)
        } else {
            setStorage('historyLog',oldStorage)
        }
    } else {
            setStorage('historyLog',emptyStorage)
    }

    // 绑定到元素 
    var logBox = $(".historyBoxForStorage")
    var g = getStorage('historyLog')
    var createLi = ''
    for (var i = 0; i < g.length; i++) {
        createLi += '<li><a class="blue" href="'+ g[i].url +'">' + g[i].tit + '</a></li>'
    }
    logBox.append(createLi)
}
    createStorageForHistory ()
})
